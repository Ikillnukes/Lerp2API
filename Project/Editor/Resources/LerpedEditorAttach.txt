#if UNITY_EDITOR

	using UnityEngine;
	using UnityEditor;
	using Lerp2API;

	[ExecuteInEditMode]
	[InitializeOnLoad]
    public class LerpedEditorHook : MonoBehaviour
    {
        static LerpedEditorHook()
        {
            Lerp2EditorMono.AddElement<LerpedEditorHook>();
        }
        void OnDestroy()
        {
			if(!EditorApplication.isPlayingOrWillChangePlaymode) 
			{
				LerpedCore.SetBool(LerpedCore.UnityBoot, false);
			}
        }
    }

#endif
